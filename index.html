                        def GeoDist_km(lat1, lon1, lat2, lon2, **kwarg):
                            R = 6371.0088
                            lat1,lon1,lat2,lon2 = map(np.radians, [lat1,lon1,lat2,lon2])
                            dlon = lon2 - lon1
                            dlat = lat2 - lat1
                            a = np.sin(dlat / 2)**2 + np.cos(lat1) * np.cos(lat2) * np.sin(dlon / 2)**2
                            c = 2 * np.arctan2(np.sqrt(a), np.sqrt(1 - a))
                        return R * c
